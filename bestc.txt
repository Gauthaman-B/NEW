int add_item()
{
    struct inventory
    {
        char item_name[20];
        int item_price;
        int item_id;
        int item_quantity;
    };
    struct inventory item[50];
    int n,i;
    printf("Enter number of item: ");
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        printf("Enter Item %d Name: ",i+1);
        scanf("%s", item[i].item_name);
        printf("Enter Item ID: ");
        scanf("%d", &item[i].item_id);
        printf("Enter Quantity: ");
        scanf("%d",&item[i].item_quantity);
        printf("Enter Item Price: ");
        scanf("%d", &item[i].item_price);
        printf("--------------------------------------------------\n");
    }
    return 0;
}

int total_inv()
{
    struct inventory
    {
        char item_name[20];
        int item_price;
        int item_id;
        int item_quantity;
    };
    struct inventory item[50];
   int n, i, tot;
   printf("*******TOTAL INVENTORY*******\n");
   for (int i = 0; i<n; i++) {
        tot += item[i].item_price*item[i].item_quantity;
    }
    printf("The total worth of inventory =Rs. %d \n",tot);
    printf("--------------------------------------------------------");
    return 0;
}
